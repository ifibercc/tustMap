options
  • 目前支持7个参数
    var polylineOptions = {
        strokeColor: String, // 线色，默认值'#3366FF'
        strokeOpacity: Number, // 线透明度，默认值1
        strokeWeight: Number, // 线宽，默认值5
        strokeStyle: String, // 线的样式，默认值'solid'
        strokeDasharray: Array, // 默认值[10, 5]
        extData: any, // 扩展属性，可以为id
        _click: Function, // 点击线的事件函数
        _dblclick: Function // 双击线的事件函数
    };
