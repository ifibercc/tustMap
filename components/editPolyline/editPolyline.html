map.editPolyline(id, options)
  • 编辑线段
    myMap.editPolyline('1234');
options
  • 目前支持2个参数
    var options = {
        _save: Function, // 保存按钮的click方法，默认值null
        _reset: Function // 重置按钮的click方法，默认值null
    };
_save
  • 通过map.findOverlays(id)方法获取polyline最后的状态，然后执行保存操作，获取到的为数组
    function _Event(e){
        var polyline = myMap.findOverlays('1314');
        console.info(polyline);
    }
_reset
  • 通过map.findOverlays(id)方法获取polyline最后的状态，然后执行保存操作
    function _Event(e){
        var marker = myMap.findOverlays('1314');
        console.info(marker);
    }
