  • 画多边形

options
  • 目前支持7个参数
    var options = {
        strokeColor: String, // 边线颜色，默认值'#FF33FF'
        strokeOpacity: Number, // 边线透明度，默认值0.2
        strokeWeight: Number, // 边线宽，默认值3
        fillColor: String, // 填充色，默认值'#1791fc'
        fillOpacity: Number, // 填充透明度，默认值0.35
        extData: any, // 扩展属性，可以为id，默认值null
        _draw: Function // 多边形画完的事件，默认值null
    };
